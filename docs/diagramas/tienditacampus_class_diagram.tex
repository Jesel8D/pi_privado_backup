\begin{class}[text width=4.5cm]{Usuario}{0,0}

% -------------------------
% Entidad Product
% -------------------------
\begin{class}[text width=4.5cm]{Producto}{10,0}
  \attribute{- id : UUID}
  \attribute{- name : String}

  \attribute{- price : Decimal}
  \attribute{- category : String}
  \attribute{- isPerishable : Boolean}
  \attribute{- currentStock : Int}
  \operation{+ updateStock(amount : Int) : void}
  \operation{+ checkPerishability() : Boolean}
\end{class}

  \attribute{- type : InventoryTypeEnum}
  \attribute{- quantity : Int}
  \attribute{- reason : String}
  \attribute{- recordedAt : Date}

\end{class}

  \attribute{- pickupLocation : String}
  \attribute{- createdAt : Date}
  \operation{+ calculateTotal() : void}
  \operation{+ updateStatus(newStatus) : void}
